        let currentSet = 1; // 1 = First Set, 2 = Second Set
        let videoElements = [];
        let currentVideoUrls = [...firstSetVideos];
        
        function showToast(message) {
            const toast = document.createElement('div');
            toast.className = 'toast';
            toast.innerHTML = `<i class="fas fa-check-circle"></i> ${message}`;
            document.body.appendChild(toast);
            setTimeout(() => toast.remove(), 2000);
        }
        
        function convertToEmbedUrl(url) {
            // Convert YouTube URL to embed URL
            let videoId = null;
            
            // Handle youtu.be format
            if (url.includes('youtu.be/')) {
                const match = url.match(/youtu\.be\/([^?&]+)/);
                if (match) videoId = match[1];
            }
            // Handle youtube.com/watch?v= format
            else if (url.includes('youtube.com/watch')) {
                const match = url.match(/[?&]v=([^&]+)/);
                if (match) videoId = match[1];
            }
            // Handle youtube.com/embed/ format
            else if (url.includes('youtube.com/embed/')) {
                const match = url.match(/embed\/([^?&]+)/);
                if (match) videoId = match[1];
            }
            
            if (videoId) {
                return `https://www.youtube.com/embed/${videoId}?enablejsapi=1&autoplay=0&rel=0`;
            }
            
            // If not YouTube, return original URL
            return url;
        }
        
        function createVideoPlayer(index, videoUrl) {
            const card = document.createElement('div');
            card.className = 'video-card';
            card.setAttribute('data-index', index);
            
            const embedUrl = convertToEmbedUrl(videoUrl);
            const isYouTube = embedUrl.includes('youtube.com/embed');
            
            card.innerHTML = `
                <div class="video-header">
                    <span class="video-title"><i class="fas fa-tv"></i> Player ${index + 1}</span>
                    <span class="video-number">#${String(index + 1).padStart(2, '0')}</span>
                </div>
                <div class="video-container" id="videoContainer_${index}">
                    ${isYouTube ? 
                        `<iframe id="video_${index}" src="${embedUrl}" style="width: 100%; height: 250px; border: none;" allow="accelerometer; autoplay; clipboard-write; encrypted-media; gyroscope; picture-in-picture" allowfullscreen></iframe>` :
                        `<video id="video_${index}" src="${embedUrl}" style="width: 100%; height: 250px; background: #000;" controls></video>`
                    }
                </div>
                <div class="video-controls">
                    <input type="text" class="video-url-input" id="urlInput_${index}" placeholder="Paste video URL (YouTube, MP4, etc.)" value="${videoUrl}">
                    <button class="video-play-btn" data-index="${index}" data-action="play"><i class="fas fa-play"></i> Play</button>
                    <button class="video-pause-btn" data-index="${index}" data-action="pause"><i class="fas fa-pause"></i> Pause</button>
                </div>
            `;
            
            return card;
        }
        
        function initVideoPlayers() {
            const grid = document.getElementById('videosGrid');
            grid.innerHTML = '';
            videoElements = [];
            
            for (let i = 0; i < 10; i++) {
                const player = createVideoPlayer(i, currentVideoUrls[i]);
                grid.appendChild(player);
                
                // Store video/iframe element reference
                const videoElement = document.getElementById(`video_${i}`);
                if (videoElement) {
                    videoElements.push(videoElement);
                } else {
                    videoElements.push(null);
                }
            }
            
            // Add event listeners to all buttons
            document.querySelectorAll('.video-play-btn').forEach(btn => {
                btn.addEventListener('click', (e) => {
                    const index = parseInt(btn.dataset.index);
                    playVideo(index);
                });
            });
            
            document.querySelectorAll('.video-pause-btn').forEach(btn => {
                btn.addEventListener('click', (e) => {
                    const index = parseInt(btn.dataset.index);
                    pauseVideo(index);
                });
            });
            
            document.querySelectorAll('.video-url-input').forEach((input, idx) => {
                input.addEventListener('change', (e) => {
                    updateSingleVideo(idx, e.target.value);
                });
            });
        }
        
        function playVideo(index) {
            const video = videoElements[index];
            if (video) {
                if (video.tagName === 'IFRAME') {
                    // For YouTube iframe, we need to use postMessage
                    video.contentWindow.postMessage('{"event":"command","func":"playVideo","args":""}', '*');
                } else {
                    video.play().catch(e => console.log('Play error:', e));
                }
                showToast(`Player ${index + 1} playing`);
            }
        }
        
        function pauseVideo(index) {
            const video = videoElements[index];
            if (video) {
                if (video.tagName === 'IFRAME') {
                    video.contentWindow.postMessage('{"event":"command","func":"pauseVideo","args":""}', '*');
                } else {
                    video.pause();
                }
                showToast(`Player ${index + 1} paused`);
            }
        }
        
        function playAllVideos() {
            videoElements.forEach((video, i) => {
                if (video) {
                    if (video.tagName === 'IFRAME') {
                        video.contentWindow.postMessage('{"event":"command","func":"playVideo","args":""}', '*');
                    } else {
                        video.play().catch(e => console.log(`Play error on player ${i + 1}:`, e));
                    }
                }
            });
            showToast(`Playing all ${videoElements.length} videos`);
        }
        
        function pauseAllVideos() {
            videoElements.forEach((video, i) => {
                if (video) {
                    if (video.tagName === 'IFRAME') {
                        video.contentWindow.postMessage('{"event":"command","func":"pauseVideo","args":""}', '*');
                    } else {
                        video.pause();
                    }
                }
            });
            showToast(`Paused all videos`);
        }
        
        function stopAllVideos() {
            videoElements.forEach((video, i) => {
                if (video) {
                    if (video.tagName === 'IFRAME') {
                        video.contentWindow.postMessage('{"event":"command","func":"stopVideo","args":""}', '*');
                    } else {
                        video.pause();
                        video.currentTime = 0;
                    }
                }
            });
            showToast(`Stopped all videos`);
        }
        
        function resetAllVideos() {
            loadSet(currentSet);
            showToast(`Reset to ${currentSet === 1 ? 'Set 1' : 'Set 2'}`);
        }
        
        function updateSingleVideo(index, url) {
            if (!url || url.trim() === '') return;
            
            const container = document.getElementById(`videoContainer_${index}`);
            const embedUrl = convertToEmbedUrl(url.trim());
            const isYouTube = embedUrl.includes('youtube.com/embed');
            
            if (container) {
                container.innerHTML = isYouTube ? 
                    `<iframe id="video_${index}" src="${embedUrl}" style="width: 100%; height: 250px; border: none;" allow="accelerometer; autoplay; clipboard-write; encrypted-media; gyroscope; picture-in-picture" allowfullscreen></iframe>` :
                    `<video id="video_${index}" src="${embedUrl}" style="width: 100%; height: 250px; background: #000;" controls></video>`;
                
                const newVideo = document.getElementById(`video_${index}`);
                if (newVideo) videoElements[index] = newVideo;
                showToast(`Player ${index + 1} URL updated`);
            }
        }
        
        function loadSet(setNumber) {
            currentSet = setNumber;
            const videosToLoad = setNumber === 1 ? firstSetVideos : secondSetVideos;
            currentVideoUrls = [...videosToLoad];
            
            for (let i = 0; i < 10; i++) {
                const container = document.getElementById(`videoContainer_${i}`);
                const embedUrl = convertToEmbedUrl(videosToLoad[i]);
                const isYouTube = embedUrl.includes('youtube.com/embed');
                
                if (container) {
                    container.innerHTML = isYouTube ? 
                        `<iframe id="video_${i}" src="${embedUrl}" style="width: 100%; height: 250px; border: none;" allow="accelerometer; autoplay; clipboard-write; encrypted-media; gyroscope; picture-in-picture" allowfullscreen></iframe>` :
                        `<video id="video_${i}" src="${embedUrl}" style="width: 100%; height: 250px; background: #000;" controls></video>`;
                    
                    const newVideo = document.getElementById(`video_${i}`);
                    if (newVideo) videoElements[i] = newVideo;
                }
                
                const input = document.getElementById(`urlInput_${i}`);
                if (input) input.value = videosToLoad[i];
            }
            
            const setIndicator = document.getElementById('currentSetName');
            const switchBtn = document.getElementById('switchSetBtn');
            if (setIndicator) setIndicator.textContent = setNumber === 1 ? 'Set 1' : 'Set 2';
            if (switchBtn) switchBtn.innerHTML = setNumber === 1 ? '<i class="fas fa-exchange-alt"></i> Switch to Set 2' : '<i class="fas fa-exchange-alt"></i> Switch to Set 1';
            
            showToast(`Loaded ${setNumber === 1 ? 'Set 1' : 'Set 2'} - 10 videos ready!`);
        }
        
        function switchSet() {
            const newSet = currentSet === 1 ? 2 : 1;
            loadSet(newSet);
        }
        
        // Global control event listeners
        document.getElementById('playAllBtn').addEventListener('click', playAllVideos);
        document.getElementById('pauseAllBtn').addEventListener('click', pauseAllVideos);
        document.getElementById('stopAllBtn').addEventListener('click', stopAllVideos);
        document.getElementById('resetAllBtn').addEventListener('click', resetAllVideos);
        document.getElementById('switchSetBtn').addEventListener('click', switchSet);
        
        // Initialize
        initVideoPlayers();
        
        console.log("MultiView Video Player Ready - 10 videos can play simultaneously!");
        console.log("First Set: 10 YouTube videos loaded");
        console.log("Click 'Switch' button to load second set of 10 videos");
    </script>
